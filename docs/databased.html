169    def _prepare_insert_queries(
170        self, table: str, columns: Iterable[str], values: Sequence[Iterable[Any]]
171    ) -> list[tuple[str, tuple[Any, ...]]]:
172        """Format a list of insert statements.
173
174        The returned value is a list because `values` will be broken up into chunks.
175
176        Each list element is a two tuple consisting of the parameterized query string and a tuple of values.
177        """
178        inserts: list[tuple[str, tuple[Any, ...]]] = []
179        max_row_count = 900
180        column_list = "(" + ", ".join(columns) + ")"
181        for i in range(0, len(values), max_row_count):
182            chunk = values[i : i + max_row_count]
183            placeholder = (
184                "(" + "),(".join((", ".join(("?" for _ in row)) for row in chunk)) + ")"
185            )
186            flattened_values = tuple((value for row in chunk for value in row))
187            inserts.append(
188                (
189                    f"INSERT INTO {table} {column_list} VALUES {placeholder};",
190                    flattened_values,
191                )
192            )
193        return inserts

510    def _format_table_data(self, table: str) -> str:
511        columns = self.get_columns(table)
512        rows = [tuple(row.values()) for row in self.select(table)]
513        inserts = self._prepare_insert_queries(table, columns, rows)
514        insert_strings: list[str] = []
515        indent = " " * 4
516        for insert in inserts:
517            text = insert[0]
518            sub = "^$data$based$^"
519            text = text.replace("?", sub)
520            for value in insert[1]:
521                if not value:
522                    value = ""
523                if isinstance(value, bool):
524                    value = int(value)
525                if not isinstance(value, int) and (not isinstance(value, float)):
526                    if isinstance(value, str):
527                        value = value.replace('"', "'")
528                    value = f'"{value}"'
529                text = text.replace(sub, str(value), 1)
530            for pair in [
531                ("INSERT INTO ", f"INSERT INTO\n{indent}"),
532                (") VALUES (", f")\nVALUES\n{indent}("),
533                ("),", f"),\n{indent}"),
534            ]:
535                text = text.replace(pair[0], pair[1])
536            insert_strings.append(text)
537        return "\n".join(insert_strings)
